    \draw (A) -- (B) -- (C) -- cycle;

        }
    }

    % Draw the smaller triangles

        \foreach \j in {0, ..., \i} {
            \draw (P\i\j) -- (P\i{\j+1}) -- (P{\i+1}\j) -- cycle;
        }
    }

    % Place the labels inside the triangles
    \foreach \i in {0, ..., 4} {
        \foreach \j in {0, ..., \i} {
            \node at (P\i\j) {\small $\alpha\beta\gamma$};
        }
    }

    % Draw the small triangles on the left side
    \draw (A) -- ++(0, 2) -- ++(2, 0) -- cycle;
