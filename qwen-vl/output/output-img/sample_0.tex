        \node at (\x,-1.1) {$k_{\x}= \x - 1$};
    }
    
    % Add the special case for the last node
    \node at (9,-0.3) {$z_{N+1} = d$};
    \node at (9,-0.7) {$j=N+1$};
    \node at (9,-1.1) {$k_{N}= N-1$};
