\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Define the fading color
    \definecolor{fadingcolor}{RGB}{255, 255, 255} % White

    % Draw the first circle with fading effect
    \fill[path fading=south, fill opacity=0.5, fading color=fadingcolor] (0,0) circle (1);

    % Draw the second circle with fading effect and blue fill
    \fill[path fading=south, fill opacity=0.5, fading color=fadingcolor, fill=blue] (1.5,0) circle (1);
\end{tikzpicture}
\end{document}
