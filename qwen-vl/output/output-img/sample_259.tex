    \draw[->] (A) -- (D);

    \draw[->] (C) -- (D);
    \draw[->] (B) -- (A);
    \draw[->] (C) -- (A);
    \draw[->] (D) -- (A);
    \draw[->] (D) -- (B);
    \draw[->] (D) -- (C);
