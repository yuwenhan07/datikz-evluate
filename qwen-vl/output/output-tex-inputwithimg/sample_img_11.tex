\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{shapes.geometric}

\begin{document}

\begin{tikzpicture}[scale=1.5]
    % Define node styles
    \tikzset{
        amp/.style={regular polygon, regular polygon sides=3, draw, fill=black, minimum size=1cm},
        bs/.style={rectangle, draw, fill=black, minimum width=1cm, minimum height=2cm},
        arrow/.style={->, >=stealth, thick}
    }

    % Draw the amplifier
    \node[amp] (amp) at (0,0) {Amp};

    % Draw the beam splitter
    \node[bs] (bs) at (2,0) {B.S.};

    % Draw the input arrow
    \draw[arrow] (-1,0) -- (amp.corner 1) node[left] {$\rho$};

    % Draw the output arrows
    \draw[arrow] (amp.corner 2) -- (bs.corner 1);
    \draw[arrow] (bs.corner 3) -- ++(1,0) node[right] {$\rho_{\mu}$};
    \draw[arrow] (bs.corner 4) -- ++(1,0) node[right] {$\rho_{N}$};
    \draw[dotted, thick] (bs.corner 2) -- ++(0,1) node[above] {$\rho_{1}$} -- ++(1,0);

    % Draw the dashed line
    \draw[dashed, thick] (bs.corner 2) -- ++(0,-1) node[below] {$\rho_{N}$} -- ++(1,0);
\end{tikzpicture}

\end{document}
