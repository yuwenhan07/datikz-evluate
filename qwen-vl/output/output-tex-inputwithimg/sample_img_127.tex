\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=1.5]
    % Define the coordinates of the vertices
    \coordinate (A1) at (0,0);
    \coordinate (B1) at (-1,1);
    \coordinate (C1) at (-2,0);
    \coordinate (D1) at (-1,-1);
    
    \coordinate (A2) at (1,0);
    \coordinate (B2) at (2,1);
    \coordinate (C2) at (3,0);
    \coordinate (D2) at (2,-1);
    
    \coordinate (A3) at (4,0);
    \coordinate (B3) at (5,1);
    \coordinate (C3) at (6,0);
    \coordinate (D3) at (5,-1);
    
    \coordinate (A4) at (7,0);
    \coordinate (B4) at (8,1);
    \coordinate (C4) at (9,0);
    \coordinate (D4) at (8,-1);
    
    % Draw the diamonds
    \draw (A1) -- (B1) -- (C1) -- (D1) -- cycle;
    \draw (A2) -- (B2) -- (C2) -- (D2) -- cycle;
    \draw (A3) -- (B3) -- (C3) -- (D3) -- cycle;
    \draw (A4) -- (B4) -- (C4) -- (D4) -- cycle;
    
    % Draw the connecting lines
    \draw (B1) -- (B2);
    \draw (C1) -- (C2);
    \draw (D1) -- (D2);
    \draw (B2) -- (B3);
    \draw (C2) -- (C3);
    \draw (D2) -- (D3);
    \draw (B3) -- (B4);
    \draw (C3) -- (C4);
    \draw (D3) -- (D4);
    
    % Draw the minimum zero forcing set (blue)
    \fill[blue] (B1) circle (2pt);
    \fill[blue] (C2) circle (2pt);
    \fill[blue] (D3) circle (2pt);
    
    % Draw the minimum dominating set (red dashed circles)
    \draw[red, dashed] (B2) circle (2pt);
    \draw[red, dashed] (C4) circle (2pt);
    
    % Draw the arc to connect the last diamond
    \draw (D4) to [out=-90,in=90] (A1);
\end{tikzpicture}

\end{document}
