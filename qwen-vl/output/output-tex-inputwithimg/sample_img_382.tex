  \node[main node] (X) {$X$};
  \node[main node] (Z) [right of=X] {$Z$};
  \node[main node] (S) [right of=Z] {$S$};
  \node[main node] (Y) [right of=S] {$Y$};
  \node[main node] (T) [below of=X] {$T$};

  \path[every node/.style={font=\sffamily\small}]
    (X) edge node [left] {} (Z)
        edge node [left] {} (T)
    (Z) edge node [left] {} (S)
        edge node [left] {} (T)
    (S) edge node [left] {} (Y)
        edge node [left] {} (T)
    (X) edge[bend left] node [left] {} (Y)
    (Z) edge[bend left] node [left] {} (Y)
    (S) edge[bend left] node [left] {} (Y)
    (T) edge[bend left] node [left] {} (Y);

\end{tikzpicture}
