\tikzset{
    hshs/.pic={
        \node at (0, 5) {HSHS};
        \draw (-6, 0) -- (6, 0);
        \foreach \i in {-3.5,...,-0.5}{
            \tikzmath{\x = \i; \y = -0.3 * \x * \x + 2.8;}
            \draw[] (\x,0) -- (\x, \y);
            \draw[fill] (\x, \y) circle(0.3);
        }
        \foreach \i in {0.5,...,3.5}{
            \tikzmath{\x = \i; \xn = -\x; \y = -0.3 * \xn * \xn + 2.8;}

    }
