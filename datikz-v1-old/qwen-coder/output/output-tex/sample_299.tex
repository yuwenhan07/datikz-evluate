To achieve a consistent color scheme in TikZ/PGF, you can use the `\definecolor` command to define specific colors based on a contiguous color range and sample points. You can then create sequences of these colors and make them accessible via custom commands.

Here's an example of how you can set up a color scheme:

```latex
\documentclass{standalone}
\usepackage{xcolor}

% Define a contiguous color range from red to blue
\definecolor{startColor}{RGB}{255,0,0} % Red
\definecolor{endColor}{RGB}{0,0,255}   % Blue

% Function to interpolate between two colors
\newcommand{\interpolate}[3]{
    \pgfmathsetmacro{\red}{#1.red + (#2.red - #1.red) * #3 / 100}
    \pgfmathsetmacro{\green}{#1.green + (#2.green - #1.green) * #3 / 100}
    \pgfmathsetmacro{\blue}{#1.blue + (#2.blue - #1.blue) * #3 / 100}
    \definecolor{resultColor}{RGB}{\red,\green,\blue}
}

% Sample points along the color range (e.g., 10 points)
\foreach \i in {0,...,9}{
    \interpolate{\startColor}{\endColor}{\i}
    \expandafter\xdef\csname colorPoint\i\endcsname{resultColor}
}

% Create a sequence of colors
\newcommand{\colorSequence}[1]{%
    \foreach \i in {0,...,#1-1}{
        \ifnum\i=0
            \let\currentColor=\csname colorPoint\i\endcsname
        \else
            \edef\currentColor{\csname colorPoint\i\endcsname}
        \fi
        \expandafter\edef\csname colorSeq\i\endcsname{\currentColor}
    }
}

% Initialize the color sequence with 10 colors
\colorSequence{10}
