\documentclass[tikz]{standalone}
\usetikzlibrary{fadings}

\tikzfading[inner path fading=west,east]
\tikzfading[inner path fading=east,west]

\begin{document}
\begin{tikzpicture}[blend group=multiply]
    % Define two semi-transparent rectangles
    \fill[opacity=0.5,black,path fading=west] (0,0) rectangle (2,2);
    \fill[opacity=0.5,black,path fading=east] (1,-1) rectangle (3,1);

    % Draw another set of semi-transparent rectangles in reverse order
    \fill[opacity=0.5,black,path fading=east] (-1,0) rectangle (1,2);
    \fill[opacity=0.5,black,path fading=west] (0,-1) rectangle (2,1);
\end{tikzpicture}
\end{document}
