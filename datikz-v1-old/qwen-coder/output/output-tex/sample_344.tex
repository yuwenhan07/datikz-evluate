\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Define the parametric equations for a sinusoidal curve
    \draw[domain=0:4*pi,samples=500] plot (\x,{sin(\x r)}); % Sinusoidal curve

    % Optionally, add arrows to indicate direction
    \draw[->] (0,-1) -- (0,1);
    \draw[->] (-1,0) -- (4*pi,0);

    % Add labels
    \node at (0,-1.2) {y};
    \node at (4*pi+0.2,0) {x};
\end{tikzpicture}
\end{document}
