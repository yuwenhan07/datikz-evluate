    \draw[fill=white] (0,0) rectangle (\h2,4); % Background rectangle

    \draw[fill=gray] (\h1,0) rectangle (\h2,\h2);

    % Add labels for clarity
    \node at (\h1/2,-0.5) [below] {$H_1$};
    \node at (\h2/2,-0.5) [below] {$H_2$};

    % Add labels for the heights
    \node at (0,\h1/2) [left] {$h_1 = 2$};
