\usetikzlibrary{shapes.geometric,backgrounds}

    stone/.style={fill=gray!80,rounded corners},

    sky/.style={fill=cyan!50},
    bridge/.style={stone,minimum width=2cm,minimum height=1cm},
    arch/.style={bridge,draw=black,thick,rounded corners=10pt}
}

\begin{document}
\begin{tikzpicture}[scale=0.8]

% Background elements
\begin{scope}[on background layer]
    % Sky
    \node[sky,rectangle,minimum width=\paperwidth,minimum height=\paperheight] (sky) {};
    
    % Water
    \node[water,rectangle,minimum width=\paperwidth,minimum height=2cm] (water) at (0,-2) {};
\end{scope}
