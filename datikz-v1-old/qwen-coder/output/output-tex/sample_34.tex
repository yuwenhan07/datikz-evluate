\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Background
    \fill[white] (-2,-2) rectangle (2,2);

    % Complex Formula
    \node at (0,0) [font=\large] {
        $z = f(\theta)$
    };

    % Black Dot
    \fill[black] (0,0) circle (0.05);

    % Path and Foliation
    \draw[thick] (-1.5,-1.5) -- (1.5,1.5);
    \foreach \x in {-1.5,-1,-0.5,0.5,1.5} {
        \foreach \y in {-1.5,-1,-0.5,0.5,1.5} {
            \ifthenelse{\abs{x}+\abs{y}<1.5}{
                \fill[red] (\x,\y) circle (0.03);
            }{}
        }
    }

    % Waves
    \draw[blue,domain=-2:2,samples=100] plot ({\x},{sin(\x r)}); % Sine Wave

    % Arrows
    \draw[-stealth] (0,-1.5) -- (0,-0.5);
    \draw[-stealth] (0,0.5) -- (0,1.5);
    \draw[-stealth] (-1.5,0) -- (-0.5,0);
    \draw[-stealth] (0.5,0) -- (1.5,0);
\end{tikzpicture}
\end{document}
