\documentclass[tikz,border=10pt]{standalone}
\usetikzlibrary{graphs,graphdrawing,shapes.geometric}

\begin{document}
\begin{tikzpicture}[every node/.style={draw,circle,inner sep=2pt}]
    \node (A) at (0,0) {};
    \node (B) at (2,0) {};
    \node (C) at (4,0) {};
    \node (D) at (6,0) {};
    \node (E) at (8,0) {};

    \node (F) at (1,-1) {};
    \node (G) at (3,-1) {};
    \node (H) at (5,-1) {};
    \node (I) at (7,-1) {};
    \node (J) at (9,-1) {};

    % Edges for the graph
    \foreach \i in {1,...,5} {
        \draw (A\i) -- (B\i);
        \draw (B\i) -- (C\i);
        \draw (C\i) -- (D\i);
        \draw (D\i) -- (E\i);
    }

    % Additional edges to create a mixed graph appearance
    \draw (A1) -- (F1);
    \draw (B2) -- (G2);
    \draw (C3) -- (H3);
    \draw (D4) -- (I4);
    \draw (E5) -- (J5);

    % Draw bridges using rectangles
    \draw[fill=gray] (2,-2) rectangle (4,-1.5);
    \draw[fill=gray] (6,-2) rectangle (8,-1.5);
