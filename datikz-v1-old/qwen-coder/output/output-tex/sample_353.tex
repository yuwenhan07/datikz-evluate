\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=1.5]

% Define colors
\definecolor{regular}{white}
\definecolor{cutcell}{black}
\definecolor{irregular}{gray}

% Draw the regular cell footprint (standard five-point Laplacian)
\draw[thick] (0,0) -- (2,0); % Horizontal top
\draw[thick] (0,2) -- (2,2); % Horizontal bottom
\draw[thick] (0,0) -- (0,2); % Vertical left
\draw[thick] (2,0) -- (2,2); % Vertical right

% Draw the diagonals for the five-point Laplacian
\draw[thick] (0,0) -- (2,2);
\draw[thick] (0,2) -- (2,0);

% Mark cut cells with dark shading
\fill[cutcell] (0.5,0.5) rectangle (1.5,1.5);

% Mark irregular cells with light shading
\fill[irregular] (1,1) rectangle (2,2);

% Mark regular cells with white
\node at (1,1) [inner sep=0pt, outer sep=0pt] {};
\node at (1.5,1.5) [inner sep=0pt, outer sep=0pt] {};
