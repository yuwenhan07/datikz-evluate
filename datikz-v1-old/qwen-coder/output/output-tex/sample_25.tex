\usetikzlibrary{shapes.geometric, arrows.meta}

\tikzset{
  startstop/.style={rectangle, rounded corners, minimum width=3cm, minimum height=1cm,text centered, draw=black, fill=red!30},
  process/.style={rectangle, minimum width=3cm, minimum height=1cm, text centered, draw=black, fill=orange!30},
  decision/.style={diamond, minimum width=3cm, minimum height=1cm, text centered, draw=black, fill=green!30},
  arrow/.style={thick,->,>=stealth}
}

\begin{document}

\begin{tikzpicture}[node distance=2cm]

\node (start) [startstop] {Start};
\node (input) [process, below of=start, yshift=-1cm] {Input Data};
\node (preliminary_analysis) [decision, below of=input, yshift=-2cm] {Preliminary Analysis};
\node (modeling_layer_1) [process, below of=preliminary_analysis, yshift=-3cm] {Modeling Layer 1};
\node (modeling_layer_2) [process, below of=modeling_layer_1, yshift=-4cm] {Modeling Layer 2};

\node (end) [startstop, below of=modeling_layer_3, yshift=-6cm] {End};
