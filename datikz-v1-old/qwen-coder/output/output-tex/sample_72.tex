\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{shapes.geometric, arrows.meta}

\begin{document}
\begin{tikzpicture}[node distance=2cm]
    % Draw the central white square
    \node[draw=white, fill=white, minimum size=4cm] (unitaritySquare) {};

    % Add the title "Unitarity (1)"
    \node[above=0.5cm of unitaritySquare, font=\large] {Unitarity (1)};

    % Surrounding elements
    \node[left=of unitaritySquare, text width=3cm] (eq1) {
        $W_1 = W_1^{\dagger}$
    };
    
    \node[right=of unitaritySquare, text width=3cm] (eq2) {
        $\sum_{i,j} U_{ij} U_{ji}^\dagger = I$
    };

    \node[below=of unitaritySquare, text width=3cm] (eq3) {
        $\partial_\mu W_1 + i g [A_\mu, W_1] = 0$
    };

    % Arrows connecting the equations to the square
    \draw[-Stealth] (eq1.east) -- node[midway, above] {$\Rightarrow$} (unitaritySquare.west);
    \draw[-Stealth] (eq2.west) -- node[midway, below] {$\Rightarrow$} (unitaritySquare.east);
    \draw[-Stealth] (eq3.north) -- node[midway, left] {$\Rightarrow$} (unitaritySquare.south);

    % Background color
    \fill[white] (-6,-6) rectangle (6,6);
\end{tikzpicture}
\end{document}
