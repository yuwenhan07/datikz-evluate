\documentclass{standalone}

    \def\axisstyle{\draw[->] #1; }
    \def\gridstyle{\draw[dashed, thin] #1; }

    % Set up the axis limits and ticks
    \def\xmin{-2}
    \def\xmax{4}
    \def\ymin{0}
    \def\ymax{10}
    \def\xtickstep{1}
    \def\ytickstep{2}

    % Draw the x-axis and y-axis
    \axisstyle{(0,\ymin) -- (\xmax,0)}
    \axisstyle{(\xmin,0) -- (0,\ymax)}

    % Draw the grid
    \foreach \x in {\xmin,\xtickstep,...,\xmax} {
        \gridstyle{(\x,0) -- (\x,\ymax)};
    }
    \foreach \y in {0,\ytickstep,...,\ymax} {
        \gridstyle{(0,\y) -- (\xmax,\y)};
    }

    % Label the axes
    \node at (\xmax,0.5) [below] {$Time Synchronization Constant$};
    \node at (-0.5,\ymax) [left]  {$Power Distribution$};
