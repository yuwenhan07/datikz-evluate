\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Define the positions for the dots
    \def\numbers{{0.03, 0.003, 0.01, 0.031, 0.05}}
    
    % Draw the vertical lines
    \foreach \i in {1,...,\the\numbers} {
        \draw (0, -\i*0.1) -- (1, -\i*0.1);
    }
    
    % Draw the horizontal lines connecting the dots
    \foreach \i in {1,...,\the\numbers} {
        \ifnum\i<\the\numbers
            \draw (\i*0.1, 0) -- (\i*0.1, -(\i+1)*0.1);
        \fi
    }
    
    % Place the numbers at the top of each vertical line
    \foreach \n/\x in {0.03/0.1, 0.003/0.2, 0.01/0.3, 0.031/0.4, 0.05/0.5} {
        \node at (\x, 0.1) {\small $\n$};
    }
    
    % Draw the dots
    \foreach \i in {1,...,\the\numbers} {
        \fill (0, -\i*0.1) circle (0.03);
    }
\end{tikzpicture}
\end{document}
