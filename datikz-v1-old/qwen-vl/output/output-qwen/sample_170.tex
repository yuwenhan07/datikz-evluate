\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Draw the magenta arrows
    \draw[magenta, thick] (0,0) -- (1,0);
    \draw[magenta, thick] (1,0) -- (1.5,0.5);
    
    % Draw the black connecting line
    \draw[black, thick] (1,0) -- (1.5,0.5);
    
    % Draw the geometric shape
    \draw[blue, thick] (2,0) circle (0.5);
    \foreach \angle in {0,30,...,330} {
        \draw[white, thick] (2,0) -- ({2+0.5*cos(\angle)},{0.5*sin(\angle)});
    }
    
    % Add some additional details
    \foreach \angle in {0,60,...,360} {
        \draw[blue, thick] ({2+0.5*cos(\angle)},{0.5*sin(\angle)}) -- ({2+0.75*cos(\angle)},{0.75*sin(\angle)});
    }
\end{tikzpicture}
\end{document}
