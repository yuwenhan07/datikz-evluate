\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Define the coordinates for the corners
    \coordinate (A) at (0,0);
    \coordinate (B) at (1,0);
    \coordinate (C) at (1,1);
    \coordinate (D) at (0,1);

    % Draw the rectangle
    \draw (A) -- (B) -- (C) -- (D) -- cycle;

    % Trim the top-right corner
    \draw (B) -- ++(0.5,-0.5) -- ++(-0.5,0) -- ++(0,0.5) -- cycle;
\end{tikzpicture}
\end{document}
