\begin{figure}[h]

        % States
        \node[state] (s1) {$s_1$};

        \node[state] (s3) [below right of=s1] {$s_3$};
        \node[state] (s4) [below right of=s2] {$s_4$};
        \node[state] (s5) [below right of=s3] {$s_5$};

        \path[->]

            (s2) edge node {$h$} (s4)
            (s3) edge node {$h$} (s5);
    \end{tikzpicture}
    \caption{(a) Initial state machine with states $s_1$, $s_2$, $s_3$, $s_4$, and $s_5$.}
