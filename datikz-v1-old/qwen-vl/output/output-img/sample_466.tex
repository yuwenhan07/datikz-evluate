\usepackage{tikz}
\usetikzlibrary{arrows.meta}

\begin{document}

\begin{tikzpicture}[scale=0.7]
    % Define coordinates for the vertices of the triangles
    \coordinate (A) at (0,0);
    \coordinate (B) at (4,0);
    \coordinate (C) at (2,3);

    % Draw the first triangle
    \draw[dotted] (A) -- node[above] {$x$} (B) -- node[above] {$y$} (C) -- node[above] {$z$} cycle;

    % Draw the arrow
    \draw[-{Stealth[scale=1.5]}] (4.5,1.5) -- (6,1.5);

    % Define coordinates for the vertices of the second triangle
    \coordinate (D) at (9,0);
    \coordinate (E) at (13,0);
    \coordinate (F) at (11,3);

    % Draw the second triangle
    \draw[dotted] (D) -- node[above] {$x$} (E) -- node[above] {$y$} (F) -- node[above] {$z$} cycle;
    \draw[dotted] (D) -- node[above] {$xz$} (F) -- node[above] {$yz$} (E) -- node[above] {$xy$} cycle;

    % Label the vertices of the second triangle with their products
    \node at (10.5,1.5) {$xz$};
    \node at (12.5,1.5) {$yz$};
    \node at (11.5,1.5) {$xy$};
\end{tikzpicture}
