\begin{tikzpicture}[scale=1.5]
    % Define coordinates for the vertices of the hexagon
    \foreach \i in {0,...,12} {
        \pgfmathsetmacro{\angle}{360/13 * \i}
        \coordinate (v\i) at (\angle:2);
    }
    
    % Draw the hexagon
    \draw (v0) -- (v1) -- (v2) -- (v3) -- (v4) -- (v5) -- cycle;
    \draw (v6) -- (v7) -- (v8) -- (v9) -- (v10) -- (v11) -- (v12) -- cycle;
    \draw (v0) -- (v6) -- (v12) -- (v8) -- (v4) -- (v0);
    \draw (v1) -- (v7) -- (v13) -- (v9) -- (v5) -- (v1);
    \draw (v2) -- (v8) -- (v14) -- (v10) -- (v6) -- (v2);
    \draw (v3) -- (v9) -- (v15) -- (v11) -- (v7) -- (v3);
    
    % Label the vertices
    \foreach \i in {0,...,12} {
        \node at (v\i) {\i};
    }
    
    % Draw dashed lines for the outer hexagon
    \draw[dashed] (v0) -- (v6) -- (v12) -- (v8) -- (v4) -- (v0);
    \draw[dashed] (v1) -- (v7) -- (v13) -- (v9) -- (v5) -- (v1);
    \draw[dashed] (v2) -- (v8) -- (v14) -- (v10) -- (v6) -- (v2);
    \draw[dashed] (v3) -- (v9) -- (v15) -- (v11) -- (v7) -- (v3);
    
    % Draw the inner hexagon
    \draw (v0) -- (v3) -- (v6) -- (v9) -- (v12) -- (v5) -- cycle;
    \draw (v1) -- (v4) -- (v7) -- (v10) -- (v13) -- (v8) -- cycle;
    \draw (v2) -- (v5) -- (v8) -- (v11) -- (v14) -- (v9) -- cycle;
    
    % Label the inner vertices
    \foreach \i in {0,...,5} {
        \node at (v3*\i+1.5) {\i};
    }
    
    % Draw the inner dashed lines
    \draw[dashed] (v0) -- (v3) -- (v6) -- (v9) -- (v12) -- (v5) -- cycle;
    \draw[dashed] (v1) -- (v4) -- (v7) -- (v10) -- (v13) -- (v8) -- cycle;
    \draw[dashed] (v2) -- (v5) -- (v8) -- (v11) -- (v14) -- (v9) -- cycle;
\end{tikzpicture}
