\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=0.5]
    % Define the size of the outermost square
    \def\squareSize{4cm}

    % Function to draw a square with a smaller square inside
    \newcommand{\drawNestedSquare}[3]{
        % Outer square
        \fill[black] (#1,#2) rectangle ++(#3,-#3);
        % Inner square
        \fill[white] (#1+#3/2,#2+#3/2) rectangle ++(-#3/2,-#3/2);
    }

    % Draw the first row (L shapes)
    \drawNestedSquare{-\squareSize}{-\squareSize}{\squareSize};
    \drawNestedSquare{0}{-\squareSize}{\squareSize};

    % Draw the second row (numbers)
    \drawNestedSquare{-\squareSize}{0}{\squareSize};
    \drawNestedSquare{0}{0}{\squareSize};

    % Draw additional squares around the main structure
    \foreach \x in {-\squareSize,0} {
        \foreach \y in {-\squareSize,0} {
            \ifnum\x=0\relax
                \ifnum\y=0\relax
                    % Skip the center square
                \else
                    \fill[black] (\x,\y) rectangle ++(\squareSize,-\squareSize);
                \fi
            \else
                \fill[black] (\x,\y) rectangle ++(\squareSize,-\squareSize);
            \fi
        }
    }

    % Add labels for the numbers
    \node at (-\squareSize/2,-\squareSize/2) {L};
    \node at (\squareSize/2,-\squareSize/2) {L};
    \node at (-\squareSize/2,\squareSize/2) {1};
    \node at (\squareSize/2,\squareSize/2) {2};

\end{tikzpicture}

\end{document}
