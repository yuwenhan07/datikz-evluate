\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=0.8]
    \def\R{5} % Number of terms in the sum
    \def\m#1{2} % Common difference m_k

    % Draw the grid for the partition
    \foreach \i in {1,...,\R}{
        \foreach \j in {1,...,\R-\i+1}{
            \draw[fill=blue!30] (\j-1,-\i) rectangle (\j,-\i+1);
        }
    }

    % Draw the labels for each term
    \foreach \i in {1,...,\R}{
        \node at (\R/2,-\i) [below] {$(\beta_{\i} + m_{\i-1} + 1)(R - \i + 1)$};
    }

    % Draw the total value R
    \node at (\R/2,-\R-0.5) [below] {$-R$};

    % Add vertical lines to separate the terms
    \foreach \i in {1,...,\R}{
        \draw[dashed] (0,-\i) -- (\R,-\i);
    }

    % Label the axes
    \draw[->] (-0.5,0) -- (\R+0.5,0) node[right] {$k$};
    \draw[->] (0,-\R-1) -- (0,1) node[above] {$\text{Summands}$};
\end{tikzpicture}

\end{document}
