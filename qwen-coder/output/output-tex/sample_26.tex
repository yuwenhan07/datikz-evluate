\documentclass[border=10pt]{standalone}
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=3]
    % Draw the full sphere for reference
    \draw[dashed] (0,0) circle (1);
    
    % Draw the spherical cap
    \fill[top color=blue!50,bottom color=blue!20] (0,0) -- (-1,0) arc[start angle=-90,end angle=90,radius=1] -- cycle;
    
    % Label the minimum and maximum comoving distances
    \node at (-1,-0.2) {$s_{\text{min}}$};
    \node at (1,-0.2) {$s_{\text{max}}$};
    
    % Draw the transversal cut
    \draw[thick] (0,0) -- (0.7,-0.6);
    
    % Label the effective comoving distance
    \node at (0.7,-0.8) {$s_{\text{eff}}$};
    
    % Draw the central axis
    \draw[->,thick] (0,0) -- (0,1) node[above]{$\theta_{\text{MAX}}$};
    
    % Label theta MAX
    \node at (0,1.2) {Central Axis};
    
    % Draw the angular opening
    \draw[dashed] (0,0) -- (cos(45),sin(45));
    \node at (cos(45),sin(45)+0.05) {\large$\theta_{\text{MAX}}$};
    
    % Draw the dashed line indicating theta MAX
    \draw[dashed] (0,0) -- (cos(45),-sin(45));
\end{tikzpicture}
