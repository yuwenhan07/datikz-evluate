\documentclass{standalone}
\usepackage{tikz}

\begin{document}

% Define styles for the nodes
\tikzset{
    matrix/.style={
        matrix of math nodes,
        row sep=-4pt,
        column sep=-4pt,
        text height=1.5ex,
        text depth=0.25ex
    }
}

\begin{tikzpicture}[node distance=2cm]

    % Weak Injectivity Transfer Matrix
    \node[matrix] (T_weak) {
        1 & 0 \\
        0 & 1
    };
    \node[right=of T_weak] {Weak Injectivity};

    % Strong Injectivity Transfer Matrix
    \node[matrix, right=of T_weak] (T_strong) {
        2 & 0 \\
        0 & 3
    };
    \node[right=of T_strong] {Strong Injectivity};

\end{tikzpicture}
