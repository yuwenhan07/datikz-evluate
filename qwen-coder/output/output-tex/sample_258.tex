\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Define circle styles
    \tikzset{
        mycircle/.style={draw, circle, minimum size=1cm},
        arrowhead/.style={->, >=stealth', thick}
    }

    % Draw circles and arrows
    \node[mycircle] (left) at (-2, 1) {};
    \node[mycircle] (right) at (2, 1) {};
    \node[mycircle] (middle) at (0, 0) {};

    % Add arrows to circles
    \draw[arrowhead] (left.north west) -- ++(-0.5, -0.5);
    \draw[arrowhead] (left.north east) -- ++(0.5, -0.5);
    \draw[arrowhead] (right.south west) -- ++(-0.5, 0.5);
    \draw[arrowhead] (right.south east) -- ++(0.5, 0.5);
    \draw[arrowhead] (middle.west) -- ++(-0.5, 0);
    \draw[arrowhead] (middle.east) -- ++(0.5, 0);

    % Labels for clarity
    \node[left] at (-2, 1.3) {Top Left};
    \node[right] at (2, 1.3) {Top Right};
    \node[below] at (0, -0.3) {Middle};

\end{tikzpicture}
\end{document}
