\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[node distance=2cm, auto]

    % Define nodes
    \node (consumer) [rectangle, draw] {Consumer};
    \node (preferences) [ellipse, draw, below of=consumer, yshift=-1cm] {Preferences};
    \node (distributions) [rectangle, draw, below of=preferences, yshift=-1cm] {Distributions};
    \node (mixing_distributions) [rectangle, draw, left of=distributions, xshift=-3cm] {Constituent Mixing Distributions};
    \node (logit_models) [rectangle, draw, right of=distributions, xshift=3cm] {Mixed Logit Models};
    \node (havan_model) [rectangle, draw, below of=logit_models, yshift=-1cm] {HAVAN Model};
    \node (aggregate_preferences) [ellipse, draw, below of=havan_model, yshift=-1cm] {Aggregate Preferences};

    % Draw arrows
    \draw[->] (consumer) -- node[anchor=east] {Selects Preferences} (preferences);
    \draw[->] (preferences) -- node[anchor=north] {Estimates Parameters} (distributions);
    \draw[->] (mixing_distributions) -- node[anchor=south] {Contributes} (distributions);
    \draw[->] (distributions) -- node[anchor=west] {Mixed Logit Models Estimate} (logit_models);
    \draw[->] (logit_models) -- node[anchor=north] {Predict Distribution Parameters} (havan_model);
    \draw[->] (havan_model) -- node[anchor=south] {Predicts Aggregate Preferences} (aggregate_preferences);

\end{tikzpicture}
\end{document}
