\documentclass[tikz,border=10pt]{standalone}
\usetikzlibrary{shapes.geometric, arrows.meta}

\tikzset{
    process/.style={rectangle, rounded corners, minimum width=3cm, minimum height=2cm, text centered, draw=black, fill=blue!20},
    decision/.style={diamond, minimum width=3cm, minimum height=2cm, text centered, draw=black, fill=red!20},
    line/.style={->, thick, >=stealth'}
}

\begin{document}
\begin{tikzpicture}[node distance=2cm]

    % Nodes
    \node (generation) [process] {Generation};
    \node (verification) [decision, below of=generation, yshift=-1cm] {Verification};
    \node (success) [process, right of=verification, xshift=2cm] {Success};
    \node (failure) [process, left of=verification, xshift=-2cm] {Failure};

    % Arrows
    \draw [line] (generation) -- node[anchor=south] {$q_i = q'_i$} (verification);
    \draw [line] (verification) -| node[near start, anchor=north] {Failed} (failure);
    \draw [line] (verification) -| node[near end, anchor=south] {Passed} (success);

\end{tikzpicture}
\end{document}
