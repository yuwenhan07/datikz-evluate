\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[node distance=2cm, auto]

    % Define styles for nodes
    \tikzstyle{vertex}=[circle, draw, fill=white, text centered]
    \tikzstyle{message}=[rectangle, draw, fill=yellow!50, text centered]

    % Nodes
    \node[vertex] (v1) at (0, 0) {\( v_1 \)};
    \node[vertex] (v2) at (-2, -2) {\( v_2 \)};
    \node[vertex] (v3) at (2, -2) {\( v_3 \)};

    % Messages
    \node[message] (msg1) at (1, -1) {\(+\)};
    \node[message] (msg2) at (-1, -1) {\(-\)};

    % Edges and labels
    \draw[-stealth] (v1) -- (msg1);
    \draw[-stealth] (v2) -- (msg2);
    \draw[-stealth] (msg1) -- (v3);
    \draw[-stealth] (msg2) -- (v3);

    % Labels for messages
    \node[above left=of msg1] {Message from \( v_1 \)};
    \node[above right=of msg2] {Message from \( v_2 \)};
    \node[below=of v3] {New sign: \( ? \)};
