\definecolor{filtering}{RGB}{158, 202, 225} % Light blue
\definecolor{reverb}{RGB}{255, 165, 0}      % Orange
\definecolor{combined}{RGB}{245, 222, 179}  % Beige

\begin{document}

    \node (filtering) [rectangle, draw=black, fill=filtering, rounded corners, text centered] {Filtering};
    \node (reverb) [rectangle, draw=black, fill=reverb, rounded corners, text centered, below=of filtering] {Reverb};
    \node (combined) [rectangle, draw=black, fill=combined, rounded corners, text centered, below=of reverb] {Combined};
