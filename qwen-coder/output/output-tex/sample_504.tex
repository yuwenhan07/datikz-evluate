\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Draw the first segment
    \draw[thick] (0,0) -- node[midway, below] {$a_0 = 2$} (2,0);
    
    % Draw subsequent segments
    \foreach \i in {1,...,4} {
        \pgfmathsetmacro{\length}{2 * (1/3)^(\i-1)}
        \draw[thick] (\i*2,0) -- node[midway, below] {$a_{\i-1} = \frac{2}{3^{\i-1}}$} (\i*2 + \length,0);
    }
    
    % Add labels for clarity
    \node at (0,-0.5) {$a_0$};
    \node at (2,-0.5) {$a_1$};
    \node at (4,-0.5) {$a_2$};
    \node at (6,-0.5) {$a_3$};
    
    % Add arrows to indicate the continuation
    \draw[-stealth] (8,0) -- node[right] {...} (10,0);
\end{tikzpicture}
\end{document}
