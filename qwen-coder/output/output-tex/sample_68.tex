\documentclass[tikz]{standalone}
\usetikzlibrary{graphs, graphdrawing, shapes.geometric}

\begin{document}
\begin{tikzpicture}[>=stealth]

% Initial Graph
\node (v1) at (0,0) [circle, draw] {$v_1$};
\node (v2) at (2,0) [circle, draw] {$v_2$};
\node (v3) at (4,0) [circle, draw] {$v_3$};
\draw[->] (v1) -- node[midway, above] {green} (v2);
\draw[->] (v2) -- node[midway, above] {blue} (v3);

% First Morphism: Contract Green Edge
\node (v1a) at (0,-2) [circle, draw] {$v_1'$};
\node (v3a) at (4,-2) [circle, draw] {$v_3'$};
\draw[->] (v1a) -- node[midway, above] {red} (v3a);

% Second Morphism: Contract Blue Edge
\node (v1b) at (0,-4) [circle, draw] {$v_1''$};
\node (v3b) at (4,-4) [circle, draw] {$v_3''$};
\draw[->] (v1b) -- node[midway, above] {purple} (v3b);

\end{tikzpicture}
