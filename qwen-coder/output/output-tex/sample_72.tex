\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=1.5]

% Define the coordinates for the first saddle move (start)
\node[circle, draw, inner sep=2pt] (A) at (0,0) {};
\node[circle, draw, inner sep=2pt] (B) at (2,-2) {};
\node[circle, draw, inner sep=2pt] (C) at (4,0) {};

% Draw lines for the first saddle move
\draw (A) -- (B);
\draw (B) -- (C);

% Define the coordinates for the second saddle move (end)
\node[circle, draw, inner sep=2pt] (D) at (6,0) {};
\node[circle, draw, inner sep=2pt] (E) at (8,-2) {};
\node[circle, draw, inner sep=2pt] (F) at (10,0) {};

% Draw lines for the second saddle move
\draw (D) -- (E);
\draw (E) -- (F);

% Connect the two saddle moves with a line
\draw[dashed] (C) -- (D);

\end{tikzpicture}

\end{document}
