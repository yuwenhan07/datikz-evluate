\usepackage{tikz}
\usetikzlibrary{3d,angles}

    \coordinate (O) at (0,0,0);
    \coordinate (X) at (4,0,0);
    \coordinate (Y) at (0,4,0);
    \coordinate (Z) at (0,0,4);

    % Draw the axes
    \draw[->] (O) -- (X) node[anchor=north east]{$x$};
    \draw[->] (O) -- (Y) node[anchor=south west]{$y$};
    \draw[->] (O) -- (Z) node[anchor=south east]{$z$};

    \def\x{1}

    \fill[black] (\xnew,\y,\znew) circle (0.1);
