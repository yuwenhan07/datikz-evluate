\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{shapes.geometric}

\begin{document}
\begin{tikzpicture}[node distance=2cm,
    start chain/.style={start chain=chain-\the\pgf@serial@number},
    box/.style={draw, rectangle, minimum width=1cm, minimum height=1cm, align=center},
    graybox/.style={draw, rectangle, minimum width=1cm, minimum height=1cm, align=center, fill=gray!30},
    dashedarrow/.style={->, dashed},
    normalarrow/.style={->}
]

% Nodes
\node[box] (A) {Node A};
\node[box] (B) [right=of A] {Node B};
\node[box] (C) [below=of A] {Node C};
\node[box] (D) [below=of B] {Node D};
\node[graybox] (E) [right=of C] {Node E};
\node[graybox] (F) [below=of D] {Node F};

% Transitions
\path[-stealth]
    (A) edge[normalarrow] node[above] {$t_1$} (B)
        edge[dashedarrow] node[below] {$t_2$} (C)
    (B) edge[normalarrow] node[above] {$t_3$} (D)
        edge[dashedarrow] node[below] {$t_4$} (E)
    (C) edge[dashedarrow] node[above] {$t_5$} (D)
        edge[normalarrow] node[below] {$t_6$} (E)
    (D) edge[normalarrow] node[above] {$t_7$} (F)
        edge[dashedarrow] node[below] {$t_8$} (E);

\end{tikzpicture}
\end{document}
