    % Arrow labels
    \draw[dashed] (whiteboard.east) -- node[midway, above] {$x \text{ and } y \text{ odd} \implies z \text{ odd}} (blackarrow.west);

    % Mathematical symbols on the black arrow
    \draw[->] ([yshift=-0.3cm]blackarrow.west) -- node[midway, below] {$x, y, z$} ([yshift=0.3cm]blackarrow.west);
