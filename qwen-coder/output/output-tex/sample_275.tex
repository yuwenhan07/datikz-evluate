\documentclass[tikz,border=10pt]{standalone}
\usetikzlibrary{automata, positioning}

\tikzset{
    node distance=2cm,
    every state/.style={fill=blue!30, rounded corners, minimum size=1cm},
    transition/.style={->, thick, shorten >=2pt, ->>
}

\begin{document}
\begin{tikzpicture}[font=\small]

\node[state] (IDLE) {IDLE};
\node[state, below left=of IDLE] (FETCH) {FETCH};
\node[state, below right=of IDLE] (DECODE) {DECODE};
\node[state, below=of DECODE] (EXECUTE) {EXECUTE};
\node[state, below=of EXECUTE] (WRITEBACK) {WRITEBACK};

% Transitions
\draw[transition] (IDLE) -- (FETCH);
\draw[transition] (FETCH) -- (DECODE);
\draw[transition] (DECODE) -- (EXECUTE);
\draw[transition] (EXECUTE) -- (WRITEBACK);
\draw[transition] (WRITEBACK) -- (IDLE);

% Fu and Ps connections
\node[right=of FETCH, yshift=-1cm] (FU) {Fu};
\node[left=of WRITEBACK, yshift=-1cm] (PS) {Ps};

\draw[dashed,->] (FETCH) -- (FU);
\draw[dashed,->] (EXECUTE) -- (PS);

\end{tikzpicture}
\end{document}
