\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=0.8]
    % Define the lattice points
    \foreach \i in {1,...,5} {
        \draw[gray!30] (\i,0) -- (\i,2);
    }

    % Draw the initial interface at t=0
    \foreach \i in {1,...,5} {
        \draw[thick, blue] (\i,0) -- (\i,\i);
    }
    \node[below left] at (1,0) {t=0};

    % Draw the intermediate interface at t=1
    \foreach \i in {1,...,5} {
        \draw[thick, red] (\i,0) -- (\i,2-\i);
    }
    \node[below left] at (1,0) {t=1};

    % Draw the final interface at t=2
    \foreach \i in {1,...,5} {
        \draw[thick, green] (\i,0) -- (\i,\i-1);
    }
    \node[below left] at (1,0) {t=2};
\end{tikzpicture}

\end{document}
