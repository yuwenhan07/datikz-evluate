\usepackage{amsmath}

\tikzset{
    axis/.style={->,thick},
    sphere/.style={draw=blue!50,fill=blue!20,opacity=0.7},
    rt-surface/.style={draw=red,dashed}
}

\begin{tikzpicture}[scale=2]

    % Sphere
    \node[circle,sphere] (sphere) at (0,0) {};
