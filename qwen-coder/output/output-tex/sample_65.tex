Below is a TikZ LaTeX code that generates a schematic representation of possible New Physics (NP) contributions to scalar ($S=1$) and charge-changing ($C=-1$) flavor-changing neutral current (FCNC) semileptonic processes within the two-generation limit of the Standard Model (SM). The code includes both CP-conserving and CP-violating contributions.

```latex
\documentclass[tikz,border=2mm]{standalone}
\usetikzlibrary{shapes.geometric, arrows.meta, positioning}

\tikzset{
    block/.style={rectangle, draw=black, fill=white, text centered, minimum width=3cm, minimum height=1cm},
    line/.style={->, thick, >=stealth'}
}

\begin{document}
\begin{tikzpicture}[node distance=2cm]
    
    % Nodes for the different processes
    \node (scalar_process) [block] {Scalar Process};
    \node (charge_change_process) [block, below=of scalar_process] {Charge-Change Process};

    % Nodes for the CP-conserving contributions
    \node (cp_conserve_scalar) [block, left=of scalar_process] {$|z_{C=1}|$};
    \node (cp_conserve_charge) [block, left=of charge_change_process] {$|z_{S=1}|$};

    % Nodes for the CP-violating contributions
    \node (cp_violate_scalar) [block, right=of scalar_process] {$c_I$};
    \node (cp_violate_charge) [block, right=of charge_change_process] {$c_I$};

    % Arrows connecting nodes
    \draw [line] (cp_conserve_scalar) -- node[above] {CP Conserving} (scalar_process);
    \draw [line] (cp_conserve_charge) -- node[above] {CP Conserving} (charge_change_process);
    \draw [line] (cp_violate_scalar) -- node[below] {CP Violating} (scalar_process);
    \draw [line] (cp_violate_charge) -- node[below] {CP Violating} (charge_change_process);
