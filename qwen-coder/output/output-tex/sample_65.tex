\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=2, every node/.style={draw, fill=white, inner sep=1pt}]

% Define patch size and number of bits/blocks per row/column
\def\n{3} % Patch size (number of bits/block per row/column)

% Draw the grid
\foreach \i in {0,...,\n-1} {
  \foreach \j in {0,...,\n-1} {
    \node at (\i, -\j) {};
  }
}

% Label the nodes with b_i or B_n^i
\foreach \i in {0,...,\n-1} {
  \foreach \j in {0,...,\n-1} {
    \pgfmathsetmacro{\bit}{(\i*\n)+\j+1}
    \node[below right] at (\i, -\j) {$b_{\bit}$};
  }
}

\end{tikzpicture}

\end{document}
