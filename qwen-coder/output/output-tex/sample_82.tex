\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=1]
    % Define coordinates
    \coordinate (0) at (0,0);
    \coordinate (1) at (2*\epsilon_n,0);
    \coordinate (2) at (\epsilon_n,\sqrt{3}*\epsilon_n);
    \coordinate (3) at (\epsilon_n,-\sqrt{3}*\epsilon_n);

    % Draw triangle
    \draw (0) -- (1) -- (2) -- cycle;

    % Label vertices
    \fill (0) circle (2pt) node[left] {0};
    \fill (1) circle (2pt) node[right] {1};
    \fill (2) circle (2pt) node[above right] {2};
    \fill (3) circle (2pt) node[below right] {3};

    % Draw epsilon lines for clarity
    \draw[dashed] (0) -- ++(0,1);
    \draw[dashed] (1) -- ++(-1,1);
    \draw[dashed] (1) -- ++(1,1);
    \draw[dashed] (2) -- ++(0,-1);
    \draw[dashed] (3) -- ++(0,-1);
\end{tikzpicture}
\end{document}
