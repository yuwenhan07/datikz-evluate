\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=2]

    % Define nodes for the triangle

    \node (x) at (-1,-1.732) [circle, draw, fill=white] {};
    \node (z) at (1,-1.732) [circle, draw, fill=white] {};

    % Draw edges of the triangle

    % Label vertices
    \node at (y) [left] {$y$};
    \node at (x) [below left] {$x$};
