To create a TikZ diagram that aligns the specified metrics for the given experimental settings, we can use the `tikz` package in LaTeX. Below is an example of how you might generate this diagram:

```latex
\documentclass[tikz,border=10pt]{standalone}
\usepackage{amsmath}

\begin{document}
\begin{tikzpicture}[scale=0.8]
    % Define colors for clarity
    \definecolor{blue}{RGB}{0,0,255}
    \definecolor{red}{RGB}{255,0,0}
    \definecolor{green}{RGB}{0,255,0}
    
    % Nodes for each plot
    \node[anchor=east] at (-4,2) {Left};
    \node[anchor=east] at (-4,-2) {Middle};
    \node[anchor=east] at (-4,-6) {Right};
    
    % X-axis labels
    \draw[-stealth] (-3,0) -- (4,0);
    \foreach \x in {-2,-1,0,1,2,3}
        \draw (\x,0.1) -- (\x,-0.1) node[below] {$\x$};
    
    % Y-axis labels
    \draw[-stealth] (0,-7) -- (0,3);
    \foreach \y in {-6,-5,-4,-3,-2,-1,0,1,2,3}
        \draw (0.1,\y) -- (-0.1,\y) node[left] {$\y$};
    
    % Plot Left
    \draw[blue, thick] plot coordinates {(0,2) (1,1.5) (2,1) (3,0.5)};
    
    % Plot Middle
    \draw[red, thick] plot coordinates {(0,-2) (1,-1.5) (2,-1) (3,-0.5)};
    
    % Plot Right
    \draw[green, thick] plot coordinates {(0,-6) (1,-5) (2,-4) (3,-3)};
    
    % Legends
    \node[below right] at (4,2) {Blue: Left};
    \node[below right] at (4,-2) {Red: Middle};
