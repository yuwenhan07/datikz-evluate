\documentclass{article}
\usepackage{tikz}

\begin{document}

\begin{figure}[h]
    \centering
    \begin{tikzpicture}[scale=2]
        % Define coordinates for the saddle points
        \coordinate (A) at (-1,0);
        \coordinate (B) at (1,0);
        
        % Draw the original saddle points
        \filldraw[blue] (A) circle[radius=0.05];
        \node[above left] at (A) {Saddle A};
        \filldraw[red] (B) circle[radius=0.05];
        \node[above right] at (B) {Saddle B};
        
        % Draw the mirror line (y-axis)
        \draw[dashed, thick] (0,-2) -- (0,2);
        
        % Reflect the saddle points across the y-axis
        \filldraw[green] (-A) circle[radius=0.05];
        \node[below left] at (-A) {Saddle A'};
        \filldraw[cyan] (-B) circle[radius=0.05];
        \node[below right] at (-B) {Saddle B'};
        
        % Connect the original and reflected points with dashed lines
        \draw[dashed, thick] (A) -- (-A);
        \draw[dashed, thick] (B) -- (-B);
    \end{tikzpicture}
    \caption{Consistency of Saddle Moves Under Mirror Image}
    \label{fig:saddle_mirror}
\end{figure}

\end{document}
