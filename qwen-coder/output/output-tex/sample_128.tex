\documentclass[tikz,border=5]{standalone}
\usepackage{tikz}

\begin{document}
\begin{tikzpicture}[scale=2]
    % Define the starting point
    \coordinate (start) at (0,0);
    
    % Draw the initial point
    \filldraw[black] (start) circle (2pt);
    
    % Define step size
    \def\stepsize{1cm}
    
    % Generate random walk
    \foreach \i in {1,...,10} {
        % Random direction: 0, 90, 180, or 270 degrees
        \pgfmathsetmacro{\angle}{rnd * 360 / 4};
        
        % Calculate new position
        \pgfpointadd{\pgfpointanchor{start}{center}}{\pgfpointxy{\stepsize*cos(\angle)}{\stepsize*sin(\angle)}};
        
        % Update start point to new position
        \let\start\pgfresult
        
        % Draw line to new position
        \draw[->] (start -| start) -- (\start);
        
        % Draw dot at new position
        \filldraw[black] (\start) circle (2pt);
    }
\end{tikzpicture}
\end{document}
