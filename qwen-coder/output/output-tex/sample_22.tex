Certainly! Below is the TikZ LaTeX code for the two diagrams you described:

```latex
\documentclass[tikz,border=10pt]{standalone}
\usetikzlibrary{shapes.geometric, arrows.meta}

\tikzset{
    node style/.style={draw, circle, minimum size=1cm},
    arrow style/.style={->, thick, >=Stealth[length=6pt]}
}

\begin{document}

\begin{tikzpicture}[node distance=2cm]
    % Nodes for the left diagram
    \node[node style] (S) {S};
    \node[node style, below=of S] (Nx) {$N[x]$};
    \node[node style, right=of Nx] (Ny) {$N[y]$};
    \node[node style, below=of Ny, xshift=-1cm] (Z) {$Z$};

    % Arrows for the left diagram
    \draw[arrow style] (S) -- (Nx);
    \draw[arrow style] (S) -- (Ny);
    \draw[arrow style] (Nx) -- (Z);
    \draw[arrow style] (Ny) -- (Z);

    % Labels for the left diagram
    \node[above=0.5cm of Nx] {S $\nsubseteq$ $N[x]$ \cup $N[y]$};
    \node[right=0.5cm of Z] {$|Z| \ge 5$};

    % Nodes for the right diagram
    \node[node style, at={(4,0)}] (xi) {$x_i$};
    \node[node style, above=of xi] (Y) {$Y$};
    \node[node style, below=of xi] (Ni) {$N(x_i)$};

    % Arrows for the right diagram
    \draw[arrow style] (xi) -- (Ni);
    \draw[arrow style] (Ni) -- (Y);

    % Labels for the right diagram
    \node[above=0.5cm of Ni] {Suppose, to the contrary,};
    \node[left=0.5cm of Yi] {there exists $x_i$};
    \node[right=0.5cm of Yi] {such that $|N(x_i) \cap Y| \ge 5$};
\end{tikzpicture}

\end{document}
