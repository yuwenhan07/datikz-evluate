        width=10cm,
        height=10cm,
        xlabel=$\theta_1$,
        ylabel=$\theta_2$,
        zlabel=$F(\theta)$,
        view={45}{30},
        colormap/jet,
        colorbar,
        enlargelimits=false,
        domain=-2:2,
        ydomain=-2:2,
        samples=50,
