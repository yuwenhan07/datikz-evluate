\documentclass{standalone}
\usepackage{tikz}
\usepackage{musixtex}

\begin{document}

% Set up the page with a white background
\pagecolor{white}

% Create a black cat's eye
\begin{center}
  \begin{tikzpicture}[scale=2]
    % Draw the outline of the cat's eye
    \draw (0,0) circle (1);
    
    % Fill the cat's eye with black
    \fill[black] (0,0) circle (0.8);
    
    % Add a small dot for the iris
    \fill[white] (0,0.5) circle (0.1);
    
    % Add a small dot for the pupil
    \fill[black] (0,0.4) circle (0.05);
  \end{tikzpicture}
\end{center}

\end{document}
