\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=1.5]
    % Define coordinates for the vertices
    \coordinate (u1) at (-2, 0);
    \coordinate (v1) at (-1, 1);
    \coordinate (x3) at (0, 0);
    \coordinate (v2) at (1, -1);
    \coordinate (u2) at (2, 0);

    % Draw the vertices
    \foreach \point in {u1, v1, x3, v2, u2} {
        \fill (\point) circle (2pt);
    }

    % Label the vertices
    \node at (u1) [below left] {$u_1$};
    \node at (v1) [above left] {$v_1$};
    \node at (x3) [below] {$x_3$};
    \node at (v2) [below right] {$v_2$};
    \node at (u2) [above right] {$u_2$};

    % Draw the edges
    \draw (u1) -- (v1);
    \draw (v1) -- (x3);
    \draw (x3) -- (v2);
    \draw (v2) -- (u2);

    % Highlight x3 as the middle vertex
    \draw[thick, red] (x3) circle (4pt);
\end{tikzpicture}

\end{document}
