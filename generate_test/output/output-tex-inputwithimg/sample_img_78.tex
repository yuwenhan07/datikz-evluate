        xlabel=$x$,
        ylabel=$y$,
        enlargelimits=false,
        legend pos=north east
    ]
        
        % Red curve for dμ_{α, β_1}/dx
        \addplot[red, domain=0:3, samples=50] {4 - x^2};
        \addlegendentry{$d\mu_{\alpha,\beta_1}\over dx$}
        
        % Blue curve for dμ_{α, β_2}/dx
        \addplot[blue, domain=0:4, samples=50] {-(x-2)^2 + 2};
        \addlegendentry{$d\mu_{\alpha,\beta_2}\over dx$}
        
        % Vertical dashed line at x̄
        \addplot[dashed, black, domain=0:3] {0};
        \node at (axis cs:3,-0.2) {$\bar{x}$};
        
    \end{axis}
