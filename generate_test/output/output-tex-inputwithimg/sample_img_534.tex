\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{positioning}

\begin{document}

\begin{tikzpicture}[node distance=1cm,
    vertex/.style={circle,fill=gray!80,draw,minimum size=6mm},
    leaf/.style={circle,fill=yellow,draw,minimum size=4mm}]
    
    % Define nodes for the spine of the caterpillar graph
    \node[vertex] (x1) {$x_1$};
    \node[vertex] (x2) [right=of x1] {$x_2$};
    \node[vertex] (x3) [right=of x2] {$x_3$};
    \node[vertex] (x4) [right=of x3] {$x_4$};
    \node[vertex] (x5) [right=of x4] {$x_5$};
    
    % Draw connections between the spine vertices
    \draw[thick] (x1) -- (x2);
    \draw[thick] (x2) -- (x3);
    \draw[thick] (x3) -- (x4);
    \draw[thick] (x4) -- (x5);
    
    % Define nodes for the leaves connected to each spine vertex
    \foreach \i in {1,2,...,3}
        \node[leaf] (l1\i) [below=of x1-\i] {\textcolor{black}{$\bullet$}};
    \foreach \i in {1,2,...,3}
        \node[leaf] (l2\i) [below=of x2-\i] {\textcolor{black}{$\bullet$}};
    \foreach \i in {1,2}
        \node[leaf] (l3\i) [below=of x3-\i] {\textcolor{black}{$\bullet$}};
    \node[leaf] (l41) [below=of x4] {\textcolor{black}{$\bullet$}};
    \node[leaf] (l42) [below=of x4-\i] {\textcolor{black}{$\bullet$}};
    \node[leaf] (l5) [below=of x5] {\textcolor{black}{$\bullet$}};
    
    % Connect leaves to their corresponding spine vertices
    \foreach \i in {1,2,...,3}
        \draw (x1) -- (l1\i);
    \foreach \i in {1,2,...,3}
        \draw (x2) -- (l2\i);
    \foreach \i in {1,2}
        \draw (x3) -- (l3\i);
    \draw (x4) -- (l41);
    \draw (x4) -- (l42);
    \draw (x5) -- (l5);
    
\end{tikzpicture}

\small\sf A caterpillar graph with degree sequence $(5,4^3,2,1^{11})$. Spine vertices are gray, leaves are yellow.

\end{document}
