\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta}

\begin{document}

\begin{tikzpicture}[scale=1.5]
    % Define coordinates
    \coordinate (O) at (0,0);
    \coordinate (A) at (4,0);
    \coordinate (B) at (4,3);
    \coordinate (C) at (2,3);
    \coordinate (D) at (2,6);

    % Draw axes with arrows
    \draw[-Stealth] (-1,0) -- (6,0) node[right] {$x$};
    \draw[-Stealth] (0,-1) -- (0,7) node[above] {$z$};

    % Draw wires
    \draw[thick, yellow, ->] (O) -- node[midway,below,sloped] {$s$} (2,0);
    \draw[thick, yellow, ->] (O) -- node[midway,below,sloped] {$s$} (4,0);

    % Draw measurement points
    \foreach \x in {0,2,4}
        \foreach \y [evaluate=\y as \col using int(\x+\y)] in {0,...,3}
            \fill[black!\col!red] (\x,\y) circle (0.05);

    % Draw arrow for Delta x
    \draw[thick, dashed, ->] (2.5,2) -- node[left] {$\Delta x$} ++(0.5,0);

    % Draw line distance z
    \draw[thick, dashed, ->] (2.5,3) -- node[right] {$z$} ++(-1.5,0);

    % Draw the rightmost measurement point
    \fill[black!60!yellow] (4,3) circle (0.1) node[above right] {$(x,z)$};

    % Draw vectors for B field components
    \draw[blue, ->] (4,3) -- node[midway,above right] {$x-s\cos\theta$} ++(30:1);
    \draw[blue, ->] (4,3) -- node[midway,above right,rotate=-60] {$z-s\sin\theta$} ++(-60:1);

    % Labels for wire angles
    \path (2,0) -- (4,0) node[midway,below] {$\theta$};
\end{tikzpicture}

\end{document}
