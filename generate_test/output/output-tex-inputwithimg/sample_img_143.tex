    \draw[->] (-2,0) -- (8,0) node[right] {$t$};

    % Marking times
    \foreach \x/\y in {0/0, 3/B_1, 4/B_2, 6/B_3, 7/B_4} {

        \draw[blue] (\x+0.5, 0) node[red, below] {Switch};
    }

    % Labels for initial time
    \draw (0, -0.5) node[below] {$t=0$};
