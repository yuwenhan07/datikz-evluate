\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{arrows.meta}
\usetikzlibrary{decorations.markings}

\tikzset{
  >=Stealth,
  every node/.style={fill=white},
  every label/.style={text=black},
  myarrow/.style={decorate, decoration={markings, mark=at position .5 with {\arrow{Stealth}}}}
}

\begin{document}

\begin{tikzpicture}[scale=1.5]
  % Vertices
  \coordinate (v1) at (0, 0);
  \coordinate (v2) at (-1, 1);
  \coordinate (v3) at (1, 1);

  % Triangle
  \draw (v1) -- (v2) -- (v3) -- cycle;

  % Angles
  \draw[dashed] (v1) -- ++(90:0.5) node[above] {$\psi_{v_2 v_3}$};
  \draw[dashed] (v2) -- ++(180:0.5) node[left] {$\psi_{v_1 v_2}$};
  \draw[dashed] (v3) -- ++(0:0.5) node[right] {$\psi_{v_3 v_1}$};

  % Labels
  \node at (v1) [circle, draw, inner sep=1pt, fill=black] {};
  \node at (v2) [circle, draw, inner sep=1pt, fill=black] {};
  \node at (v3) [circle, draw, inner sep=1pt, fill=black] {};

  \node at (v1) [below] {$v_1$};
  \node at (v2) [left] {$v_2$};
  \node at (v3) [right] {$v_3$};

  % Connections
  \draw[dotted] (v1) -- ++(-100:1) node[above] {$\kappa_{z_2}$};
  \draw[dotted] (v2) -- ++(-80:1) node[below] {$\kappa_{z_3}$};
  
  % Cycles
  \path (v1) edge[bend left=45, "z_1"', myarrow] node[above] {$\mathbf{C}_{v_1}$} (v2)
        (v2) edge[bend left=45, "z_2"', myarrow] node[above] {$\mathbf{C}_{v_2}$} (v3)
        (v3) edge[bend left=45, "z_3"', myarrow] node[above] {$\mathbf{C}_{v_3}$} (v1);

  % Paths
  \draw[myarrow] (v1) -- node[above] {$z_1 = \psi_{(v_1,v_2)}(z_1) = e^{i\frac{\pi}{4}} \cdot z_1$} (v2);
  \draw[myarrow] (v2) -- node[above] {$z_2 = \psi_{(v_2,v_3)}(z_2) = e^{i\frac{\pi}{4}} \cdot z_2$} (v3);
  \draw[myarrow] (v3) -- node[above] {$z_3 = \psi_{(v_3,v_1)}(z_3) = e^{i\frac{\pi}{4}} \cdot z_3$} (v1);

  % Cyclic Path
  \draw[->, thick, blue, dashed] (v1) -- ++(45:1) node[right, black] {$z_1' = \psi_C (z_1)$};

\end{tikzpicture}

\end{document}
