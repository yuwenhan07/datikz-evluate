        \node[circle, draw, fill=white, inner sep=2pt] at (\x,0) {};
    }
    
    % Draw vertical lines from nodes to the top row
    \foreach \x in {0,2,4,6,8} {
        \node[circle, draw, fill=black, inner sep=2pt] at (\x,1) {};
    }
    
    % Draw vertical lines from nodes to the bottom row
    \foreach \x in {1,3,5,7} {
        \node[circle, draw, fill=black, inner sep=2pt] at (\x,-1) {};
    }
\end{tikzpicture}

\textbf{Example of a subcubic graph which shows that the $\frac{n}{2}$-upper bound for the locating total-dominating number of a twin-free graph on $n$ vertices is not true. The shaded vertices constitute a minimum LTD-set.}
