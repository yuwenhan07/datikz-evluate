  \fi
}

  \draw[myarrow] (B\i) -- node[right] {$b_i$ instances of $X$ between $Y_i$ and $Y_{i+1}$ for $i \in \{1,\dots,a-1\}$} (B\the\numexpr\i+1\relax);
