\begin{document}
    
    \begin{tikzpicture}[thick]
        
        % Agent cell
        \node[draw,circle,inner sep=0pt,minimum size=1cm] (agent) at (-3, 2) {};

        \node[above] at (agent) {$(X_i, Y_j)$};
        
        % Exit Sign
        \node[draw,fill=green!50, rectangle, minimum width = 1 cm, minimum height=2 cm] (exit) at (3,0) {Exit};
