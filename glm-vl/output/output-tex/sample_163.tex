  \new PianoStaff {

      instrumentName = "Piano"
    } {
      \relative c' {
        c4 d4 e4 f4 g4 a4 b4
      }
    }
  }
  \layout { }
  \midi {
    \context {
      \PianoStaff
      \consists "Piano_drums_engraver"
    }
    \context {
      \DrumStaff
      \consists "Piano_drums_engraver"
    }
  }
\end{score}
